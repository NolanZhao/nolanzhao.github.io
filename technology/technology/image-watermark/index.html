int center_x = w / 2;
int center_y = h / 2;
int radius = min(center_x, center_y);
for (int y = 0; y < h; y++) {
    for (int x = 0; x < w; x++) {
        int distance = pow( y - center_y, 2) + pow(x - center_x, 2);
